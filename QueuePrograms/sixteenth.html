    void nextSmaller(int arr[]){
        int i, j;

        {

            {
                if (arr[i] > arr[j])
                {

                    break;
                }
            }
            if (j == 5 || i == 4)
            {
                printf("%d -----> -1\n", arr[i]);
            }
        }
    }
