    void nextGreater(int arr[]){
        int i, j;
        for (i = 0; i < 5; i++)
        {
            for (j = i+1; j < 5; j++)
            {
                if (arr[i] < arr[j])
                {
                    printf("%d -----> %d\n", arr[i], arr[j]);
                    break;
                }
            }
            if (j == 5 || i == 4)
            {
                printf("%d -----> -1\n", arr[i]);
            }
        }
    }
